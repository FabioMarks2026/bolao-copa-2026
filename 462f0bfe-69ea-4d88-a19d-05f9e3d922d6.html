<script>

// CONFIGURAÇÃO DOS JOGOS

const GAMES = {

1: { date: '2026-06-13', time: '19:00', home: 'Brasil', away: 'Marrocos', homeFlag: '🇧🇷', awayFlag: '🇲🇦' },

2: { date: '2026-06-19', time: '21:30', home: 'Brasil', away: 'Haiti', homeFlag: '🇧🇷', awayFlag: '🇭🇹' },

3: { date: '2026-06-24', time: '19:00', home: 'Escócia', away: 'Brasil', homeFlag: '🏴󠁧󠁢󠁳󠁣󠁴󠁿', awayFlag: '🇧🇷' }

};


const LOCK_MINUTES = 15; // minutos antes do jogo para bloquear

const BET_VALUE = 5; // R$ por palpite


// GERA ID ÚNICO DO DISPOSITIVO

function getDeviceId() {

let id = localStorage.getItem('bolao_device_id');

if (!id) {

id = 'dev_' + Math.random().toString(36).substr(2, 9) + '_' + Date.now();

localStorage.setItem('bolao_device_id', id);

}

return id;

}


// VERIFICA SE ESTÁ BLOQUEADO

function isLocked(gameId) {

const game = GAMES[gameId];

const gameDateTime = new Date(game.date + 'T' + game.time + ':00-03:00'); // Brasília UTC-3

const lockTime = new Date(gameDateTime.getTime() - LOCK_MINUTES * 60000);

const now = new Date();

return now >= lockTime;

}


// TEMPO RESTANTE PARA BLOQUEIO

function getTimeToLock(gameId) {

const game = GAMES[gameId];

const gameDateTime = new Date(game.date + 'T' + game.time + ':00-03:00');

const lockTime = new Date(gameDateTime.getTime() - LOCK_MINUTES * 60000);

const now = new Date();

const diff = lockTime - now;


if (diff <= 0) return null;


const hours = Math.floor(diff / 3600000);

const minutes = Math.floor((diff % 3600000) / 60000);

const seconds = Math.floor((diff % 60000) / 1000);


if (hours > 0) return `${hours}h ${minutes}min`;

return `${minutes}min ${seconds}s`;

}


// FORMATA DATA/HORA

function formatDateTime(date) {

const d = new Date(date);

return d.toLocaleString('pt-BR', {

day: '2-digit', month: '2-digit', year: 'numeric',

hour: '2-digit', minute: '2-digit', second: '2-digit'

});

}


// ATUALIZA CONTADOR DO COFRINHO

function updateCofrinho() {

const allPredictions = JSON.parse(localStorage.getItem('bolao_predictions') || '{}');

const deviceId = getDeviceId();

const myPredictions = allPredictions[deviceId] || {};


let totalSaved = 0;

let participantName = localStorage.getItem('bolao_participant_name') || '';


// Conta todos os palpites salvos deste dispositivo

for (let g = 1; g <= 3; g++) {

if (myPredictions[g] && myPredictions[g].saved) {

totalSaved += BET_VALUE;

}

}


document.getElementById('cofrinhoValue').textContent = 'R$ ' + totalSaved.toFixed(2).replace('.', ',');

document.getElementById('participantsCount').textContent =

(participantName ? participantName + ' - ' : '') +

totalSaved + ' palpite(s) salvo(s)';

}


// VERIFICA E ATUALIZA ESTADO DOS JOGOS

function checkGameStatus() {

for (let g = 1; g <= 3; g++) {

const btn = document.getElementById('btn-g' + g);

const info = document.getElementById('info-g' + g);

const lockMsg = document.getElementById('lock-g' + g);

const homeInput = document.getElementById('g' + g + '-home');

const awayInput = document.getElementById('g' + g + '-away');


if (isLocked(g)) {

// BLOQUEADO

btn.disabled = true;

btn.textContent = '🔒 BLOQUEADO';

homeInput.disabled = true;

awayInput.disabled = true;

lockMsg.style.display = 'block';

lockMsg.innerHTML = '⏰ Palpites encerrados!<br>15 min antes do jogo';


// Mostra palpite salvo se existir

const saved = getSavedPrediction(g);

if (saved) {

homeInput.value = saved.home;

awayInput.value = saved.away;

info.innerHTML = `✅ Seu palpite: <span class="timestamp">${saved.home} x ${saved.away}</span><br>Salvo em: ${formatDateTime(saved.timestamp)}`;

}

} else {

// ABERTO

const timeLeft = getTimeToLock(g);

if (timeLeft) {

lockMsg.style.display = 'block';

lockMsg.innerHTML = `⏰ Bloqueio em: <span class="countdown">${timeLeft}</span>`;

lockMsg.style.background = 'rgba(255,215,0,0.2)';

lockMsg.style.borderColor = '#ffd700';

lockMsg.style.color = '#ffd700';

} else {

lockMsg.style.display = 'none';

}


// Verifica se já tem palpite salvo

const saved = getSavedPrediction(g);

if (saved) {

homeInput.value = saved.home;

awayInput.value = saved.away;

info.innerHTML = `✅ Palpite salvo: <span class="timestamp">${saved.home} x ${saved.away}</span><br>em ${formatDateTime(saved.timestamp)}<br>📝 Você pode alterar até o bloqueio!`;

btn.textContent = '💾 ATUALIZAR PALPITE';

} else {

info.innerHTML = '';

btn.textContent = '💾 SALVAR PALPITE (R$ 5,00)';

}

}

}

}


// PEGA PALPITE SALVO

function getSavedPrediction(gameId) {

const allPredictions = JSON.parse(localStorage.getItem('bolao_predictions') || '{}');

const deviceId = getDeviceId();

const myPredictions = allPredictions[deviceId] || {};

return myPredictions[gameId] || null;

}


// SALVAR PALPITE

function savePrediction(gameId) {

const participantName = document.getElementById('participantName').value.trim();


if (!participantName) {

alert('⚠️ Por favor, digite seu nome antes de salvar o palpite!');

document.getElementById('participantName').focus();

return;

}


if (isLocked(gameId)) {

alert('🔒 Palpites para este jogo estão bloqueados! (15 min antes do início)');

return;

}


const homeScore = document.getElementById('g' + gameId + '-home').value;

const awayScore = document.getElementById('g' + gameId + '-away').value;


if (homeScore === '' || awayScore === '') {

alert('⚠️ Preencha ambos os placares!');

return;

}


const homeNum = parseInt(homeScore);

const awayNum = parseInt(awayScore);


if (isNaN(homeNum) || isNaN(awayNum) || homeNum < 0 || awayNum < 0 || homeNum > 20 || awayNum > 20) {

alert('⚠️ Placar inválido! Use números de 0 a 20.');

return;

}


// Salva nome do participante

localStorage.setItem('bolao_participant_name', participantName);


// Salva palpite

const allPredictions = JSON.parse(localStorage.getItem('bolao_predictions') || '{}');

const deviceId = getDeviceId();


if (!allPredictions[deviceId]) {

allPredictions[deviceId] = {};

}


const wasAlreadySaved = allPredictions[deviceId][gameId] && allPredictions[deviceId][gameId].saved;


allPredictions[deviceId][gameId] = {

home: homeNum,

away: awayNum,

timestamp: new Date().toISOString(),

saved: true,

participant: participantName,

game: GAMES[gameId]

};


localStorage.setItem('bolao_predictions', JSON.stringify(allPredictions));


// Efeito visual

createConfetti();


// Atualiza interface

updateCofrinho();

checkGameStatus();


const action = wasAlreadySaved ? 'atualizado' : 'salvo';

const game = GAMES[gameId];

alert(`✅ Palpite ${action} com sucesso!


${game.homeFlag} ${game.home} ${homeNum} x ${awayNum} ${game.away} ${game.awayFlag}


💰 Valor adicionado ao cofrinho: R$ ${BET_VALUE},00


📱 Guarde este link! Seu palpite ficará salvo neste dispositivo.`);

}


// EFEITO CONFETE

function createConfetti() {

const colors = ['#ffd700', '#009c3b', '#ffffff', '#ff6b6b', '#4ecdc4'];

for (let i = 0; i < 50; i++) {

const confetti = document.createElement('div');

confetti.className = 'confetti';

confetti.style.left = Math.random() * 100 + 'vw';

confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];

confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';

confetti.style.borderRadius = Math.random() > 0.5 ? '50%' : '0';

document.body.appendChild(confetti);

setTimeout(() => confetti.remove(), 4000);

}

}


// CARREGA DADOS SALVOS

function loadSavedData() {

const participantName = localStorage.getItem('bolao_participant_name');

if (participantName) {

document.getElementById('participantName').value = participantName;

}


for (let g = 1; g <= 3; g++) {

const saved = getSavedPrediction(g);

if (saved) {

document.getElementById('g' + g + '-home').value = saved.home;

document.getElementById('g' + g + '-away').value = saved.away;

}

}


updateCofrinho();

checkGameStatus();

}


// ATUALIZA A CADA SEGUNDO

setInterval(() => {

checkGameStatus();

}, 1000);


// INICIALIZAÇÃO

document.addEventListener('DOMContentLoaded', () => {

loadSavedData();


// Salva nome ao digitar

document.getElementById('participantName').addEventListener('input', (e) => {

localStorage.setItem('bolao_participant_name', e.target.value.trim());

});


// Previne scroll em inputs number no mobile

document.querySelectorAll('.score-input').forEach(input => {

input.addEventListener('touchstart', (e) => {

e.stopPropagation();

});

});

});


// Previne zoom no double-tap

let lastTouchEnd = 0;

document.addEventListener('touchend', (e) => {

const now = Date.now();

if (now - lastTouchEnd <= 300) {

e.preventDefault();

}

lastTouchEnd = now;

}, false);

</script>
